# Required for UK keyboard
remapblock
    BACK_QUOTE BACKSLASH
    BACKSLASH EUROPE_2
endblock

# Swap Left Ctrl & Alt around so they're closer to where I expect
remapblock
    LCTRL LALT
    LALT LCTRL
endblock

# Caps Lock -> Windows key
remapblock
    CAPS_LOCK RGUI
endblock

# Scroll Lock -> Enable select 1
ifselect any
remapblock
    SCROLL_LOCK SELECT_1
endblock

# Select 1: numpad -> Arrows
ifselect 1
remapblock
    PAD_1 END
    PAD_2 DOWN
    PAD_3 PAGE_DOWN
    PAD_4 LEFT
    PAD_6 RIGHT
    PAD_7 HOME
    PAD_8 UP
    PAD_9 PAGE_UP
    PAD_PLUS ENTER
endblock
