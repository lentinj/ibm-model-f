# Required for UK keyboard
remapblock
    BACK_QUOTE BACKSLASH
    BACKSLASH EUROPE_2
endblock

# Swap Left Ctrl & Alt around so they're closer to where I expect
remapblock
    LCTRL LALT
    LALT LCTRL
endblock

# Caps Lock -> Windows key
remapblock
    CAPS_LOCK RGUI
endblock

# Mangled numpad
remapblock
    # Pad asterix == print screen
    PAD_ASTERIX LEFT
    PAD_1 DOWN
    PAD_2 RIGHT
    PAD_3 PRINTSCREEN
    PAD_4 UP
    PAD_5 END
    PAD_6 PAGE_DOWN
    PAD_7 BACK_QUOTE
    PAD_8 HOME
    PAD_9 PAGE_UP
    SCROLL_LOCK PAUSE
    PAD_MINUS F11
    PAD_PLUS F12
    PAD_0 INSERT
    PAD_PERIOD DELETE
endblock

remapblock

endblock
